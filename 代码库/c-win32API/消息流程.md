# 消息回调函数

LRESULT CALLBACK WinPro(
HWND hwnd,

UINT msg,

WPARAM wParam,

LPARAM lParam)；

hwnd - 触发消息的窗口句柄
msg - 窗口所触发的消息
wParam - 额外参数
lParam - 额外参数
